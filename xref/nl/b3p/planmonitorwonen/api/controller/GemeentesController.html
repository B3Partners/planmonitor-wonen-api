10  
11  @RestController
12  @Profile("!test")
13  public class GemeentesController {
14    private final JdbcClient jdbcClient;
15  
16    public GemeentesController(JdbcClient jdbcClient) {
17      this.jdbcClient = jdbcClient;
18    }
19  
20    @GetMapping(path = "${planmonitor-wonen-api.base-path}/gemeentes")
21    public ResponseEntity<List<Gemeente>> planregistraties() {
22      List<Gemeente> gemeentes =
23          jdbcClient
24              .sql(
25                  "select identificatie, naam, provincie, st_astext(geometry) as geometry from gemeente order by naam")
26              .query(Gemeente.class)
27              .list();
28      return ResponseEntity.ok(gemeentes);
29    }
30  }
