7   package nl.b3p.planmonitorwonen.api.controller;
8   
9   import java.util.List;
10  import nl.b3p.planmonitorwonen.api.model.Gemeente;
11  import org.springframework.context.annotation.Profile;
12  import org.springframework.http.ResponseEntity;
13  import org.springframework.jdbc.core.simple.JdbcClient;
14  import org.springframework.web.bind.annotation.GetMapping;
15  import org.springframework.web.bind.annotation.RequestParam;
16  import org.springframework.web.bind.annotation.RestController;
17  
18  @RestController
19  @Profile("!test")
20  public class GemeentesController {
21    private final JdbcClient jdbcClient;
22  
23    public GemeentesController(JdbcClient jdbcClient) {
24      this.jdbcClient = jdbcClient;
25    }
26  
27    @GetMapping(path = "${planmonitor-wonen-api.base-path}/gemeentes")
28    public ResponseEntity<List<Gemeente>> planregistraties(@RequestParam(required = false) String provincie) {
29      JdbcClient.StatementSpec statementSpec;
30      if (provincie == null) {
31        statementSpec = jdbcClient.sql("select identificatie, naam, provincie from gemeente order by naam");
32      } else {
33        statementSpec = jdbcClient
34            .sql("select identificatie, naam, provincie from gemeente where provincie = ? order by naam")
35            .param(provincie);
36      }
37      List<Gemeente> gemeentes = statementSpec.query(Gemeente.class).list();
38      return ResponseEntity.ok(gemeentes);
39    }
40  }
