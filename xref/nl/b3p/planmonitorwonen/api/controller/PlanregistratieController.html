7   package nl.b3p.planmonitorwonen.api.controller;
8   
9   import static org.springframework.http.HttpStatus.BAD_REQUEST;
10  import static org.springframework.http.HttpStatus.NOT_FOUND;
11  import static org.springframework.http.HttpStatus.OK;
12  
13  import java.util.Map;
14  import java.util.Set;
15  import nl.b3p.planmonitorwonen.api.PlanmonitorWonenDatabaseService;
16  import nl.b3p.planmonitorwonen.api.model.Planregistratie;
17  import nl.b3p.planmonitorwonen.api.model.PlanregistratieComplete;
18  import org.locationtech.jts.io.ParseException;
19  import org.springframework.context.annotation.Profile;
20  import org.springframework.http.ResponseEntity;
21  import org.springframework.web.bind.annotation.DeleteMapping;
22  import org.springframework.web.bind.annotation.GetMapping;
23  import org.springframework.web.bind.annotation.PathVariable;
24  import org.springframework.web.bind.annotation.PutMapping;
25  import org.springframework.web.bind.annotation.RequestBody;
26  import org.springframework.web.bind.annotation.RestController;
27  import org.springframework.web.server.ResponseStatusException;
28  
29  @RestController
30  @Profile("!test")
31  public class PlanregistratieController {
32    private final PlanmonitorWonenDatabaseService pmwDb;
33  
34    public PlanregistratieController(
35        PlanmonitorWonenDatabaseService planmonitorWonenDatabaseService) {
36      this.pmwDb = planmonitorWonenDatabaseService;
37    }
38  
39    @GetMapping(path = "${planmonitor-wonen-api.base-path}/planregistraties")
40    public Set<Planregistratie> planregistraties() {
41      return pmwDb.getPlanregistraties();
42    }
43  
44    @GetMapping(path = "${planmonitor-wonen-api.base-path}/planregistratie/{id}/details")
45    public Map<String, Object> details(@PathVariable("id") String id) {
46      if (!pmwDb.planregistratieExists(id)) {
47        throw new ResponseStatusException(NOT_FOUND);
48      }
49      return Map.of(
50          "plancategorieen",
51          pmwDb.getPlancategorieen(id),
52          "detailplanningen",
53          pmwDb.getDetailplanningen(id));
54    }
55  
56    @PutMapping(path = "${planmonitor-wonen-api.base-path}/planregistratie/{id}")
57    public ResponseEntity<?> put(
58        @PathVariable("id") String id, @RequestBody PlanregistratieComplete planregistratieComplete)
59        throws ParseException {
60      if (id == null || !id.equals(planregistratieComplete.planregistratie().getId())) {
61        throw new ResponseStatusException(BAD_REQUEST);
62      }
63      pmwDb.insertPlanregistratie(
64          planregistratieComplete.planregistratie(),
65          planregistratieComplete.plancategorieen(),
66          planregistratieComplete.detailplanningen());
67      return ResponseEntity.status(OK).build();
68    }
69  
70    @DeleteMapping(path = "${planmonitor-wonen-api.base-path}/planregistratie/{id}")
71    public ResponseEntity<?> delete(@PathVariable("id") String id) {
72      if (id == null) {
73        return ResponseEntity.status(BAD_REQUEST).build();
74      } else if (!pmwDb.planregistratieExists(id)) {
75        return ResponseEntity.status(NOT_FOUND).build();
76      } else {
77        pmwDb.deletePlanregistratie(id);
78        return ResponseEntity.status(OK).build();
79      }
80    }
81  }
