7   package nl.b3p.planmonitorwonen.api.controller;
8   
9   import java.util.Map;
10  import nl.b3p.planmonitorwonen.api.security.PlanmonitorAuthenticationService;
11  import org.springframework.context.annotation.Profile;
12  import org.springframework.http.MediaType;
13  import org.springframework.security.core.annotation.CurrentSecurityContext;
14  import org.springframework.security.core.context.SecurityContext;
15  import org.springframework.web.bind.annotation.GetMapping;
16  import org.springframework.web.bind.annotation.RestController;
17  
18  @RestController
19  @Profile("!test")
20  public class HelloController {
21    private final PlanmonitorAuthenticationService planmonitorAuthenticationService;
22  
23    public HelloController(PlanmonitorAuthenticationService planmonitorAuthenticationService) {
24      this.planmonitorAuthenticationService = planmonitorAuthenticationService;
25    }
26  
27    @GetMapping(path = "${planmonitor-wonen-api.base-path}/hello", produces = MediaType.APPLICATION_JSON_VALUE)
28    public Map<String, Object> hello(@CurrentSecurityContext SecurityContext securityContext) {
29      PlanmonitorAuthenticationService.PlanmonitorAuthentication auth =
30          planmonitorAuthenticationService.getFromSecurityContext();
31  
32      return Map.of(
33          "name",
34          securityContext.getAuthentication().getName(),
35          "authorities",
36          securityContext.getAuthentication().getAuthorities().stream()
37              .map(Object::toString)
38              .toList(),
39          "isProvincie",
40          auth.isProvincie(),
41          "gemeentes",
42          auth.gemeentes());
43    }
44  }
