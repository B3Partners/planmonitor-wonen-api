7   package nl.b3p.planmonitorwonen.api.controller;
8   
9   import static org.springframework.web.bind.annotation.RequestMethod.GET;
10  
11  import com.fasterxml.jackson.databind.node.ObjectNode;
12  import java.io.Serializable;
13  import java.lang.invoke.MethodHandles;
14  import nl.b3p.planmonitorwonen.api.security.TMAPIAuthenticationToken;
15  import org.slf4j.Logger;
16  import org.slf4j.LoggerFactory;
17  import org.springframework.http.HttpStatus;
18  import org.springframework.http.MediaType;
19  import org.springframework.http.ResponseEntity;
20  import org.springframework.security.core.annotation.AuthenticationPrincipal;
21  import org.springframework.security.core.context.SecurityContextHolder;
22  import org.springframework.security.core.userdetails.UserDetails;
23  import org.springframework.web.bind.annotation.RequestMapping;
24  import org.springframework.web.bind.annotation.RestController;
25  
26  @RestController
27  @RequestMapping(
28      path = "${planmonitor-wonen-api.base-path}/hello",
29      produces = MediaType.APPLICATION_JSON_VALUE)
30  public class HelloController {
31    private static final Logger logger =
32        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
33  
34    private record HelloResponse(String message, ObjectNode authResponse) implements Serializable {}
35  
36    @RequestMapping(
37        method = {GET},
38        produces = MediaType.APPLICATION_JSON_VALUE)
39    public ResponseEntity<Serializable> hello(@AuthenticationPrincipal UserDetails userDetails) {
40      final TMAPIAuthenticationToken authentication =
41          (TMAPIAuthenticationToken) SecurityContextHolder.getContext().getAuthentication();
42  
43      logger.info("Authentication: {}", authentication);
44      logger.info("UserDetails: {}", userDetails);
45  
46      if (null == authentication) {
47        // should not happen
48        return ResponseEntity.status(HttpStatus.UNAUTHORIZED)
49            .body(new HelloResponse("Unauthorized", null));
50      }
51  
52      return ResponseEntity.status(HttpStatus.OK)
53          .body(
54              new HelloResponse(
55                  "Hello "
56                      + authentication.getName()
57                      + " with roles: "
58                      + authentication.getAuthorities()
59                      + " and userDetails: "
60                      + userDetails,
61                  authentication.getAuthResponse()));
62    }
63  }
