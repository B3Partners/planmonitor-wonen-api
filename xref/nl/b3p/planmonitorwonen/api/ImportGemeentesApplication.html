30  public class ImportGemeentesApplication {
31    @SuppressWarnings("PMD.SystemPrintln")
32    public static void main(String[] args) throws IOException {
33      System.out.println("Loading gemeentes from WFS...");
34      String wfs = "https://service.pdok.nl/kadaster/bestuurlijkegebieden/wfs/v1_0";
35      String typename = "bestuurlijkegebieden:Gemeentegebied";
36      String sql = getGemeentesSql(wfs, typename);
37      new OutputStreamWriter(new FileOutputStream("gemeentes.sql"), UTF_8).append(sql).close();
38      System.out.println("Done");
39    }
40  
41    public static String getGemeentesSql(String wfs, String typename) throws IOException {
42      Map<String, Object> params = new HashMap<>();
43      params.put(WFSDataStoreFactory.URL.key, wfs);
44      DataStore ds = DataStoreFinder.getDataStore(params);
45  
46      SimpleFeatureSource fs = ds.getFeatureSource(typename);
47      StringWriter sw = new StringWriter();
48      try (SimpleFeatureIterator features = fs.getFeatures().features();
49          PrintWriter pw = new PrintWriter(sw)) {
50        while (features.hasNext()) {
51          SimpleFeature feature = features.next();
52  
53          String sql =
54              "insert into gemeente(identificatie, naam, provincie, geometry) values('%s', '%s', '%s', '%s');\n";
55          Geometry g = (Geometry) feature.getDefaultGeometry();
56          g = TopologyPreservingSimplifier.simplify(g, 50);
57          g = new GeometryPrecisionReducer(new PrecisionModel(1.0)).reduce(g);
58          String wkb = WKBWriter.toHex(new WKBWriter().write(g)).trim();
59          pw.printf(
60              sql,
61              escapeSql(feature.getAttribute("identificatie")),
62              escapeSql(feature.getAttribute("naam")),
63              escapeSql(feature.getAttribute("ligtInProvincieNaam")),
64              wkb);
65        }
66      }
67  
68      return sw.toString();
69    }
70  
71    private static String escapeSql(Object s) {
72      return s.toString().replaceAll("'", "''");
73    }
