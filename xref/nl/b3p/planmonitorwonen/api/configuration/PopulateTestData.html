34    private static final Logger logger =
35        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
36

38  
39    private final JdbcClient jdbcClient;

41    public PopulateTestData(PlanmonitorWonenDatabaseService pmwDb, JdbcClient jdbcClient) {
42      this.pmwDb = pmwDb;
43      this.jdbcClient = jdbcClient;

46    @PostConstruct

48      if (!pmwDb.getPlanregistratiesForProvincie().isEmpty()) {
49        return;

51  
52      boolean gemeentesLoaded = false;
53      try {
54        logger.info("Laden gemeentes WFS...");
55        populateGemeentes();
56        logger.info(
57            "Succes, {} gemeentes geladen",
58            this.jdbcClient.sql("select count(*) from gemeente").query().singleValue());
59        gemeentesLoaded = true;
60      } catch (Exception e) {
61        logger.error("Fout bij laden van gemeentes", e);
62      }
63  
64      String wkt =
65          "POLYGON ((27791.55972066056 397785.16112066095, 24697.205902478643 393790.63164628064, 26778.86210743738 391483.93152727233, 30548.347667768074 395084.6341520658, 30548.347667768074 395084.6341520658, 27791.55972066056 397785.16112066095))";
66  
67      Planregistratie planregistratie = new Planregistratie()
68          .setId(UUID.randomUUID().toString())
69          .setGeometrie(wkt)
70          .setCreator("test")
71          .setCreatedAt(OffsetDateTime.now(ZoneId.of("Europe/Amsterdam")))
72          .setPlanNaam("Plan 1")
73          .setProvincie("Zeeland")
74          .setGemeente(gemeentesLoaded ? "Middelburg" : null)

76          .setPlaatsnaam("Middelburg")
77          .setVertrouwelijkheid("Openbaar")
78          .setOpdrachtgeverType("Gemeente")
79          .setOpdrachtgeverNaam("De opdrachtgever")
80          .setOpmerkingen("Opmerking\nTweede regel")
81          .setPlantype("Herstructurering")
82          .setBestemmingsplan("Een bestemmingsplan")
83          .setStatusProject("Voorbereiding")

86          .setBeoogdWoonmilieuAbf13("Centrum")

126   @Value("${planmonitor-wonen-api.wfs.bestuurlijke-gebieden}")
127   private String bestuurlijkeGebiedenWfs;
128 
129   @Value("${planmonitor-wonen-api.wfs.bestuurlijke-gebieden.gemeentes-typename}")
130   private String gemeentesTypename;
131 
132   private void populateGemeentes() throws IOException {
133     String sql = ImportGemeentesApplication.getGemeentesSql(bestuurlijkeGebiedenWfs, gemeentesTypename);
134     // Intentionally clear the gemeente table before (re)importing test/demo data.
135     // This configuration is only active when the `planmonitor-wonen-api.populate-testdata`
136     // property is set to `true` and is not meant to be enabled in production environments.
137     // Ensure that this remains disabled in production and that schema constraints allow
138     // truncating and repopulating the gemeente table in this way.
139     this.jdbcClient.sql("delete from gemeente").update();
140     this.jdbcClient.sql(sql).update();
141   }
