13  import java.time.ZoneId;
14  import java.util.UUID;

24  import org.springframework.context.annotation.Configuration;

27  
28  @Configuration
29  @ConditionalOnProperty(name = "planmonitor-wonen-api.populate-testdata", havingValue = "true")
30  @Profile("!test")
31  public class PopulateTestData {
32    private static final Logger logger =
33        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
34  
35    private final PlanmonitorWonenDatabaseService pmwDb;
36  
37    private final JdbcClient jdbcClient;
38  
39    public PopulateTestData(PlanmonitorWonenDatabaseService pmwDb, JdbcClient jdbcClient) {
40      this.pmwDb = pmwDb;
41      this.jdbcClient = jdbcClient;
42    }
43  
44    @PostConstruct
45    public void init() throws ParseException {
46      if (!pmwDb.getPlanregistraties().isEmpty()) {
47        return;
48      }
49  
50      boolean gemeentesLoaded = false;
51      try {
52        logger.info("Laden gemeentes WFS...");
53        populateGemeentes();
54        logger.info(
55            "Succes, {} gemeentes geladen",
56            this.jdbcClient.sql("select count(*) from gemeente").query().singleValue());
57        gemeentesLoaded = true;
58      } catch (Exception e) {
59        logger.error("Fout bij laden van gemeentes", e);
60      }
61  
62      String wkt =
63          "POLYGON ((27791.55972066056 397785.16112066095, 24697.205902478643 393790.63164628064, 26778.86210743738 391483.93152727233, 30548.347667768074 395084.6341520658, 30548.347667768074 395084.6341520658, 27791.55972066056 397785.16112066095))";
64  
65      Planregistratie planregistratie =
66          new Planregistratie()
67              .setId(UUID.randomUUID().toString())
68              .setGeometrie(wkt)
69              .setCreator("test")
70              .setCreatedAt(OffsetDateTime.now(ZoneId.of("Europe/Amsterdam")))
71              .setPlanNaam("Plan 1")
72              .setProvincie("Zeeland")
73              .setGemeente(gemeentesLoaded ? "Middelburg (Z.)" : null)
74              .setRegio("Walcheren")
75              .setPlaatsnaam("Middelburg")
76              .setVertrouwelijkheid("Openbaar")
77              .setOpdrachtgeverType("Gemeente")
78              .setOpdrachtgeverNaam("De opdrachtgever")
79              .setOpmerkingen("Opmerking\nTweede regel")
80              .setPlantype("Herstructurering")
81              .setBestemmingsplan("Een bestemmingsplan")
82              .setStatusProject("Voorbereiding")
83              .setStatusPlanologisch("3. In voorbereiding")
84              .setKnelpuntenMeerkeuze("Bereikbaarheid")
85              .setBeoogdWoonmilieuAbf13("Centrum-dorps")
86              .setAantalStudentenwoningen(70)
87              .setSleutelproject(false);
88      Plancategorie[] plancategorieen = {
89        new Plancategorie(
90            UUID.randomUUID().toString(),
91            planregistratie.getId(),
92            null,
93            null,
94            null,
95            null,
96            "Nieuwbouw",
97            null,
98            null,
99            null,
100           null,
101           null,
102           20,
103           0)
104     };
105     Detailplanning[] detailplanningen = {
106       new Detailplanning(
107           UUID.randomUUID().toString(), plancategorieen[0].id(), null, null, null, null, 2025, 10),
108       new Detailplanning(
109           UUID.randomUUID().toString(), plancategorieen[0].id(), null, null, null, null, 2028, 10)
110     };
111     pmwDb.insertPlanregistratie(planregistratie, plancategorieen, detailplanningen);
112   }
113 
114   private void populateGemeentes() throws IOException {
115     String sql = ImportGemeentesApplication.getGemeentesSql();
116     this.jdbcClient.sql(sql).update();
117   }
118 }
