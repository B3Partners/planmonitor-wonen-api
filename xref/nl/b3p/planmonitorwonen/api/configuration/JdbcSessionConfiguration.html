42  @Configuration
43  @EnableJdbcHttpSession
44  @Profile("!test")
45  public class JdbcSessionConfiguration implements BeanClassLoaderAware {
46    private static final Logger logger =
47        LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
48  
49    @Value("${spring.datasource.url}")
50    private String dataSourceUrl;
51  
52    @Value("${spring.datasource.username}")
53    private String dataSourceUsername;
54  
55    @Value("${spring.datasource.password}")
56    private String dataSourcePassword;
57  
58    @Value("${tailormap.datasource.url}")
59    private String sessionDataSourceUrl;
60  
61    @Value("${tailormap.datasource.username}")
62    private String sessionDataSourceUsername;
63  
64    @Value("${tailormap.datasource.password}")
65    private String sessionDataSourcePassword;
66  
67    private static final String CREATE_SESSION_ATTRIBUTE_QUERY =
68        """
69  INSERT INTO %TABLE_NAME%_ATTRIBUTES (SESSION_PRIMARY_ID, ATTRIBUTE_NAME, ATTRIBUTE_BYTES)
70  VALUES (?, ?, convert_from(?, 'UTF8')::jsonb)
71  """;
72  
73    private static final String UPDATE_SESSION_ATTRIBUTE_QUERY =
74        """
75  UPDATE %TABLE_NAME%_ATTRIBUTES
76  SET ATTRIBUTE_BYTES = encode(?, 'escape')::jsonb
77  WHERE SESSION_PRIMARY_ID = ?
78  AND ATTRIBUTE_NAME = ?
79  """;
80  
81    private ClassLoader classLoader;
82  
83    @Override
84    public void setBeanClassLoader(@NonNull ClassLoader classLoader) {
85      this.classLoader = classLoader;
86    }
87  
88    @Bean
89    public DataSource dataSource() {
90      DataSourceBuilder<?> dataSourceBuilder = DataSourceBuilder.create();
91      dataSourceBuilder.url(dataSourceUrl);
92      dataSourceBuilder.username(dataSourceUsername);
93      dataSourceBuilder.password(dataSourcePassword);
94      return dataSourceBuilder.build();
95    }
96  
97    @Bean
98    public JdbcClient jdbcClient(DataSource dataSource) {
99      return JdbcClient.create(dataSource);
100   }
101 
102   @Bean(name = "tailormapJdbcClient")
103   public JdbcClient tailormapJdbcClient(@Qualifier("tailormapDataSource") DataSource data) {
104     return JdbcClient.create(data);
105   }
106 
107   @Bean
108   SessionRepositoryCustomizer<JdbcIndexedSessionRepository> customizer() {
109     return (sessionRepository) -> {
110       sessionRepository.setCreateSessionAttributeQuery(CREATE_SESSION_ATTRIBUTE_QUERY);
111       sessionRepository.setUpdateSessionAttributeQuery(UPDATE_SESSION_ATTRIBUTE_QUERY);
112     };
113   }
114 
115   @Bean(name = {"springSessionDataSource", "tailormapDataSource"})
116   @SpringSessionDataSource
117   public DataSource sessionDataSource() {
118     DataSourceBuilder<?> dataSourceBuilder = DataSourceBuilder.create();
119     dataSourceBuilder.url(sessionDataSourceUrl);
120     dataSourceBuilder.username(sessionDataSourceUsername);
121     dataSourceBuilder.password(sessionDataSourcePassword);
122     return dataSourceBuilder.build();
123   }
124 
125   @Bean(name = "springSessionTransactionOperations")
126   public PlatformTransactionManager springSessionTransactionOperations(
127       @Qualifier("springSessionDataSource") DataSource springSessionDatasource) {
128     return new DataSourceTransactionManager(springSessionDatasource);
129   }
130 
131   @Bean("springSessionConversionService")
132   public ConversionService springSessionConversionService() {
133     BasicPolymorphicTypeValidator.Builder builder = BasicPolymorphicTypeValidator.builder()
134         .allowIfSubType("org.tailormap.api.security.")
135         .allowIfSubType("org.springframework.security.")
136         .allowIfSubType("java.util.")
137         .allowIfSubType(java.lang.Number.class)
138         .allowIfSubType("java.time.")
139         .allowIfBaseType(Object.class);
140 
141     JsonMapper mapper = JsonMapper.builder()
142         .configure(
143             StreamReadFeature.INCLUDE_SOURCE_IN_LOCATION,
144             (logger.isDebugEnabled() || logger.isTraceEnabled()))
145         .configure(SerializationFeature.INDENT_OUTPUT, (logger.isDebugEnabled() || logger.isTraceEnabled()))
146         .addMixIn(
147             org.tailormap.api.security.TailormapUserDetailsImpl.class,
148             org.tailormap.api.security.TailormapUserDetailsImplMixin.class)
149         .addMixIn(
150             org.tailormap.api.security.TailormapOidcUser.class,
151             org.tailormap.api.security.TailormapOidcUserMixin.class)
152         .addModules(SecurityJacksonModules.getModules(this.classLoader, builder))
153         .activateDefaultTyping(builder.build(), DefaultTyping.NON_FINAL, JsonTypeInfo.As.PROPERTY)
154         .build();
155 
156     final GenericConversionService converter = new GenericConversionService();
157     // Object -> byte[] (serialize to JSON bytes)
158     // this is not actually done in the application because sessions are
159     // read-only, so this is commented out but left here for possible future use
160     //    converter.addConverter(Object.class, byte[].class, source -> {
161     //      try {
162     //        logger.debug("Serializing Spring Session: {}", source);
163     //        return mapper.writerFor(Object.class).writeValueAsBytes(source);
164     //      } catch (JacksonException e) {
165     //        logger.error("Error serializing Spring Session object: {}", source, e);
166     //        throw new ConversionFailedException(
167     //            TypeDescriptor.forObject(source), TypeDescriptor.valueOf(byte[].class), source, e);
168     //      }
169     //    });
170     // byte[] -> Object (deserialize from JSON bytes)
171     converter.addConverter(byte[].class, Object.class, source -> {
172       try {
173         logger.debug(
174             "Deserializing Spring Session from bytes, length: {} ({})",
175             source.length,
176             new String(source, StandardCharsets.UTF_8));
177         return mapper.readValue(source, Object.class);
178       } catch (JacksonException e) {
179         String preview;
180         try {
181           String content = new String(source, StandardCharsets.UTF_8);
182           int maxLength = 256;
183           if (logger.isDebugEnabled() || logger.isTraceEnabled()) {
184             preview = content;
185           } else {
186             preview = content.length() > maxLength ? content.substring(0, maxLength) + "..." : content;
187           }
188         } catch (Exception ex) {
189           preview = "<unavailable>";
190         }
191         logger.error(
192             "Error deserializing Spring Session from bytes, length: {}, preview: {}",
193             source.length,
194             preview,
195             e);
196         throw new ConversionFailedException(
197             TypeDescriptor.valueOf(byte[].class), TypeDescriptor.valueOf(Object.class), source, e);
198       }
199     });
200 
201     return converter;
202   }
