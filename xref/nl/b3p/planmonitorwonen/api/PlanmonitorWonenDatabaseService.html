37  @Service

39  public class PlanmonitorWonenDatabaseService {
40    private final JdbcClient jdbcClient;
41    private final RowMapper<Planregistratie> planregistratieRowMapper;
42  
43    public PlanmonitorWonenDatabaseService(JdbcClient jdbcClient) {
44      this.jdbcClient = jdbcClient;
45  
46      final GenericConversionService conversionService = new GenericConversionService();
47      DefaultConversionService.addDefaultConverters(conversionService);
48      conversionService.addConverter(
49          new Converter<PgArray, String[]>() {
50            @Override
51            public String[] convert(PgArray source) {
52              try {
53                return (String[]) source.getArray();
54              } catch (SQLException e) {
55                throw new RuntimeException(e);
56              }
57            }
58          });
59  
60      planregistratieRowMapper =
61          new SimplePropertyRowMapper<>(Planregistratie.class, conversionService) {
62            @Override
63            @NonNull
64            public Planregistratie mapRow(@NonNull ResultSet rs, int rowNumber) throws SQLException {
65              Planregistratie planregistratie = super.mapRow(rs, rowNumber);
66              planregistratie.setGeometrie(wkbToWkt(planregistratie.getGeometrie()));
67              return planregistratie;
68            }
69          };
70    }
71  
72    private String wkbToWkt(String wkb) {
73      if (wkb == null) {
74        return null;
75      }
76      try {
77        Geometry geometry = new WKBReader().read(WKBReader.hexToBytes(wkb));
78        return new WKTWriter().write(geometry);
79      } catch (ParseException e) {
80        throw new RuntimeException(e);
81      }
82    }
83  
84    private byte[] wktToWkb(String wkt) throws ParseException {
85      Geometry geometry = new WKTReader().read(wkt);
86      return new WKBWriter().write(geometry);
87    }
88  
89    private String sqlQuestionMarks(int count) {
90      String[] qs = new String[count];
91      Arrays.fill(qs, "?");
92      return String.join(", ", qs);
93    }
94  
95    public Set<Planregistratie> getPlanregistratiesForProvincie() {
96      return jdbcClient.sql("select * from planregistratie").query(planregistratieRowMapper).set();
97    }

99    public Set<Planregistratie> getPlanregistratiesForGemeentes(Collection<String> gemeentes) {
100     return jdbcClient
101         .sql(
102             "select * from planregistratie where gemeente in (%s)"
103                 .formatted(sqlQuestionMarks(gemeentes.size())))
104         .params(Arrays.asList(gemeentes.toArray()))
105         .query(planregistratieRowMapper)
106         .set();
107   }

114   @Transactional
115   public void insertPlanregistratie(
116       Planregistratie planregistratie,
117       List<Plancategorie> plancategorieen,
118       List<Detailplanning> detailplanningen)
119       throws ParseException {
120     this.deletePlanregistratie(planregistratie.getId());
121     String insertPlanregistratie =
122 """
123 insert into planregistratie(
124   id,
125   geometrie,
126   creator,
127   created_at,
128   editor,
129   edited_at,
130   plan_naam,
131   provincie,
132   gemeente,
133   regio,
134   plaatsnaam,
135   vertrouwelijkheid,
136   opdrachtgever_type,
137   opdrachtgever_naam,
138   opmerkingen,
139   plantype,
140   bestemmingsplan,
141   status_project,
142   status_planologisch,
143   knelpunten_meerkeuze,
144   beoogd_woonmilieu_abf13,
145   aantal_studentenwoningen,
146   sleutelproject
147   )
148 values (%s, ?::pmw_knelpunten_meerkeuze[], %s)"""
149             .formatted(sqlQuestionMarks(19), sqlQuestionMarks(3));
150     this.jdbcClient
151         .sql(insertPlanregistratie)
152         .param(1, planregistratie.getId(), Types.OTHER)
153         .param(wktToWkb(planregistratie.getGeometrie()))
154         .param(planregistratie.getCreator())
155         .param(planregistratie.getCreatedAt())
156         .param(planregistratie.getEditor())
157         .param(planregistratie.getEditedAt())
158         .param(planregistratie.getPlanNaam())
159         .param(planregistratie.getProvincie())
160         .param(planregistratie.getGemeente())
161         .param(planregistratie.getRegio())
162         .param(planregistratie.getPlaatsnaam())
163         .param(12, planregistratie.getVertrouwelijkheid(), Types.OTHER)
164         .param(13, planregistratie.getOpdrachtgeverType(), Types.OTHER)
165         .param(planregistratie.getOpdrachtgeverNaam())
166         .param(planregistratie.getOpmerkingen())
167         .param(16, planregistratie.getPlantype(), Types.OTHER)
168         .param(planregistratie.getBestemmingsplan())
169         .param(18, planregistratie.getStatusProject(), Types.OTHER)
170         .param(19, planregistratie.getStatusPlanologisch(), Types.OTHER)
171         .param(20, planregistratie.getKnelpuntenMeerkeuze(), Types.ARRAY)
172         .param(21, planregistratie.getBeoogdWoonmilieuAbf13(), Types.OTHER)
173         .param(planregistratie.getAantalStudentenwoningen())
174         .param(planregistratie.isSleutelproject())
175         .update();
176 
177     for (Plancategorie p : plancategorieen) {
178       this.jdbcClient
179           .sql(
180               """
181             insert into plancategorie(id, planregistratie_id, creator, created_at, editor, edited_at, nieuwbouw, woning_type, wonen_en_zorg, flexwoningen, betaalbaarheid, sloop, totaal_gepland, totaal_gerealiseerd)
182             values (%s)"""
183                   .formatted(sqlQuestionMarks(14)))
184           .param(1, p.id(), Types.OTHER)
185           .param(2, p.planregistratieId(), Types.OTHER)
186           .param(p.creator())
187           .param(p.createdAt())
188           .param(p.editor())
189           .param(p.editedAt())
190           .param(7, p.nieuwbouw(), Types.OTHER)
191           .param(8, p.woningType(), Types.OTHER)
192           .param(9, p.wonenEnZorg(), Types.OTHER)
193           .param(10, p.flexwoningen(), Types.OTHER)
194           .param(11, p.betaalbaarheid(), Types.OTHER)
195           .param(12, p.sloop(), Types.OTHER)
196           .param(p.totaalGepland())
197           .param(p.totaalGerealiseerd())
198           .update();
199     }
200     for (Detailplanning d : detailplanningen) {
201       this.jdbcClient
202           .sql(
203               """
204             insert into detailplanning(id, plancategorie_id, creator, created_at, editor, edited_at, jaartal, aantal_gepland)
205             values (%s)"""
206                   .formatted(sqlQuestionMarks(8)))
207           .param(1, d.id(), Types.OTHER)
208           .param(2, d.plancategorieId(), Types.OTHER)
209           .param(d.creator())
210           .param(d.createdAt())
211           .param(d.editor())
212           .param(d.editedAt())
213           .param(d.jaartal())
214           .param(d.aantalGepland())
215           .update();
216     }
217   }
