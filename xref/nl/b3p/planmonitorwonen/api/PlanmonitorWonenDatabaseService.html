10  import java.sql.SQLException;

30  @Service
31  @Profile("!test")
32  public class PlanmonitorWonenDatabaseService {
33    private final JdbcClient jdbcClient;
34    private final RowMapper<Planregistratie> planregistratieRowMapper;
35  
36    public PlanmonitorWonenDatabaseService(JdbcClient jdbcClient) {
37      this.jdbcClient = jdbcClient;
38  
39      planregistratieRowMapper =
40          new SimplePropertyRowMapper<>(Planregistratie.class) {
41            @Override
42            public Planregistratie mapRow(ResultSet rs, int rowNumber) throws SQLException {
43              Planregistratie planregistratie = super.mapRow(rs, rowNumber);
44              assert planregistratie != null;
45              planregistratie.setGeometrie(wkbToWkt(planregistratie.getGeometrie()));
46              return planregistratie;
47            }
48          };
49    }
50  
51    private String wkbToWkt(String wkb) {
52      if (wkb == null) {
53        return null;
54      }
55      try {
56        Geometry geometry = new WKBReader().read(WKBReader.hexToBytes(wkb));
57        return new WKTWriter().write(geometry);
58      } catch (ParseException e) {
59        throw new RuntimeException(e);
60      }
61    }
62  
63    private byte[] wktToWkb(String wkt) throws ParseException {
64      Geometry geometry = new WKTReader().read(wkt);
65      return new WKBWriter().write(geometry);
66    }
67  
68    private String sqlQuestionMarks(int count) {
69      String[] qs = new String[count];
70      Arrays.fill(qs, "?");
71      return String.join(", ", qs);
72    }
73  
74    public Set<Planregistratie> getPlanregistraties() {
75      return jdbcClient.sql("select * from planregistratie").query(planregistratieRowMapper).set();
76    }
77  
78    @Transactional
79    public void deletePlanregistratie(String id) {
80      jdbcClient.sql("delete from planregistratie where id = ?").param(1, id, Types.OTHER).update();
81    }
82  
83    @Transactional
84    public void insertPlanregistratie(
85        Planregistratie planregistratie,
86        Plancategorie[] plancategorieen,
87        Detailplanning[] detailplanningen)
88        throws ParseException {
89      this.deletePlanregistratie(planregistratie.getId());
90      String insertPlanregistratie =
91              """
92  insert into planregistratie(
93    id,
94    geometrie,
95    creator,
96    created_at,
97    editor,
98    edited_at,
99    plan_naam,
100   provincie,
101   gemeente,
102   regio,
103   plaatsnaam,
104   vertrouwelijkheid,
105   opdrachtgever_type,
106   opdrachtgever_naam,
107   opmerkingen,
108   plantype,
109   bestemmingsplan,
110   status_project,
111   status_planologisch,
112   knelpunten_meerkeuze,
113   beoogd_woonmilieu_abf13,
114   aantal_studentenwoningen,
115   sleutelproject
116   )
117 values (%s)"""
118             .formatted(sqlQuestionMarks(22));
119     this.jdbcClient
120         .sql(insertPlanregistratie)
121         .param(1, planregistratie.getId(), Types.OTHER)
122         .param(wktToWkb(planregistratie.getGeometrie()))
123         .param(planregistratie.getCreator())
124         .param(planregistratie.getCreatedAt())
125         .param(planregistratie.getEditor())
126         .param(planregistratie.getEditedAt())
127         .param(planregistratie.getPlanNaam())
128         .param(planregistratie.getProvincie())
129         .param(planregistratie.getGemeente())
130         .param(planregistratie.getRegio())
131         .param(planregistratie.getPlaatsnaam())
132         .param(12, planregistratie.getVertrouwelijkheid(), Types.OTHER)
133         .param(13, planregistratie.getOpdrachtgeverType(), Types.OTHER)
134         .param(planregistratie.getOpdrachtgeverNaam())
135         .param(planregistratie.getOpmerkingen())
136         .param(16, planregistratie.getPlantype(), Types.OTHER)
137         .param(planregistratie.getBestemmingsplan())
138         .param(18, planregistratie.getStatusProject(), Types.OTHER)
139         .param(19, planregistratie.getStatusPlanologisch(), Types.OTHER)
140         .param(20, planregistratie.getKnelpuntenMeerkeuze(), Types.OTHER)
141         .param(21, planregistratie.getBeoogdWoonmilieuAbf13(), Types.OTHER)
142         .param(planregistratie.getAantalStudentenwoningen())
143         .param(planregistratie.isSleutelproject())
144         .update();
145 
146     for (Plancategorie p : plancategorieen) {
147       this.jdbcClient
148           .sql(
149                   """
150             insert into plancategorie(id, planregistratie_id, creator, created_at, editor, edited_at, nieuwbouw, woning_type, wonen_en_zorg, flexwoningen, betaalbaarheid, sloop, totaal_gepland, totaal_gerealiseerd)
151             values (%s)"""
152                   .formatted(sqlQuestionMarks(14)))
153           .param(1, p.id(), Types.OTHER)
154           .param(2, p.planregistratieId(), Types.OTHER)
155           .param(p.creator())
156           .param(p.createdAt())
157           .param(p.editor())
158           .param(p.editedAt())
159           .param(7, p.nieuwbouw(), Types.OTHER)
160           .param(8, p.woningType(), Types.OTHER)
161           .param(9, p.wonenEnZorg(), Types.OTHER)
162           .param(10, p.flexwoningen(), Types.OTHER)
163           .param(11, p.betaalbaarheid(), Types.OTHER)
164           .param(12, p.sloop(), Types.OTHER)
165           .param(p.totaalGepland())
166           .param(p.totaalGerealiseerd())
167           .update();
168     }
169     for (Detailplanning d : detailplanningen) {
170       this.jdbcClient
171           .sql(
172                   """
173             insert into detailplanning(id, plancategorie_id, creator, created_at, editor, edited_at, jaartal, aantal_gepland)
174             values (%s)"""
175                   .formatted(sqlQuestionMarks(8)))
176           .param(1, d.id(), Types.OTHER)
177           .param(2, d.plancategorieId(), Types.OTHER)
178           .param(d.creator())
179           .param(d.createdAt())
180           .param(d.editor())
181           .param(d.editedAt())
182           .param(d.jaartal())
183           .param(d.aantalGepland())
184           .update();
185     }
186   }

188   public boolean planregistratieExists(String id) {
189     return !this.jdbcClient
190         .sql("select 1 from planregistratie where id = ?")
191         .param(1, id, Types.OTHER)
192         .query()
193         .singleColumn()
194         .isEmpty();
195   }

200         .param(1, planregistratieId, Types.OTHER)
201         .query(Plancategorie.class)

203   }

205   public Set<Detailplanning> getDetailplanningen(String planregistratieId) {
206     return this.jdbcClient
207         .sql(
208             """
209             select * from detailplanning
210             where plancategorie_id in (select id from plancategorie where planregistratie_id = ?)""")
211         .param(1, planregistratieId, Types.OTHER)
212         .query(Detailplanning.class)
213         .set();
214   }
